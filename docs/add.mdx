``` dart
extension CiC on String {
  static final numberFormat = NumberFormat("#,##0.00", "en_US");
  String toCiCNumber() {
    var numb = double.tryParse(this);
    if (numb != null) {
      return numberFormat.format(numb);
    } else {
      debugPrint('here');
      return this;
    }
  }
}
```
